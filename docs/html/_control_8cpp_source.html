157  ep2 = currTarget.position.y - currPose.position.y;
158  // Avoiding problems with divide by zero

161  }
162

165

172  ROS_INFO_STREAM("ep3: " << ep3);
173 
174  // calculating distance to target
175  distToTarget = std::sqrt(
176  std::pow(ep1, 2) + std::pow(ep2, 2) + std::pow(ep3, 2));
177  setPtX = std::sqrt(std::pow(ep1, 2) + std::pow(ep2, 2));
178 
179  // Removing point from target after getting near target
180  if (distToTarget < 0.5) {
181  if (travPath.empty()) {
182  if (fabs(currPose.position.x) < 0.5 && fabs(currPose.position.y) < 0.5) {
183  landPub.publish(emptyMsg);
184  ros::shutdown();

195  }
196  } else {

198  if (travPath.empty()) {

201  landPub.publish(emptyMsg);

212  travPath.push_back(p);
213  }
214  }
215  }
216  }
217  // getting pose in RPY from quaternion
218  geometry_msgs::Point currRPY = quat2RPY(currPose);
219  double M_theta;
220  // Angle from X axis using ArcTan2
221  M_theta = atan2(ep2, ep1);
222 
223  // Calculating absolute difference
224  double yaw = currRPY.z;
225  // Trignometry to calculate the difference in angle
226  double Mag = std::sqrt(std::pow(ep1, 2) + std::pow(ep2, 2));
227  double AbsDiff = acos((ep1 * cos(yaw) + ep2 * sin(yaw)) / Mag);
228  double YawDiffVal = 0.15;
229 
230  // Angle between orientation of drone and target vector
231  ROS_INFO_STREAM("AbsDiff: " << AbsDiff);
232 
233  // Calculating the error for yaw using trignometry
234  if (AbsDiff < 1.50) {
235  // Getting turning direction using
236  // cross product numerator
237  double direction = (ep1 * sin(yaw) - ep2 * cos(yaw));
238  int sign = 1;
239  if (direction < 0) {

241  }
242  YawDiffVal = -1 * sign * AbsDiff;
243  } else {

245  }
246

249  // Displaying error in rotation

251  // time from previous iteration

253  // time after previous iteration

256

261  pidZ.setDt(dt);
262  pidYaw.setDt(dt);
263
